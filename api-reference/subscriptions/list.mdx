---
title: "Listar assinaturas"
api: "GET /v1/subscriptions"
description: "Lista paginada (cursor) de assinaturas da tenant."
---

**Escopo:** `subscriptions:read`

## Query params

<ParamField query="per_page" type="integer" default="25">1-100.</ParamField>
<ParamField query="cursor" type="string">Cursor.</ParamField>
<ParamField query="status" type="string">`active`, `trialing`, `past_due`, `canceled`.</ParamField>
<ParamField query="user_id" type="integer">Filtra por user.</ParamField>

## Response

<ResponseField name="data[].id" type="integer">ID.</ResponseField>
<ResponseField name="data[].user_id" type="integer">FK do user.</ResponseField>
<ResponseField name="data[].checkout_id" type="integer|null">FK do checkout.</ResponseField>
<ResponseField name="data[].gateway" type="string">`stripe`, `mercadopago`, `asaas`.</ResponseField>
<ResponseField name="data[].status" type="string">Status atual.</ResponseField>
<ResponseField name="data[].amount" type="string">Valor decimal.</ResponseField>
<ResponseField name="data[].currency" type="string">ISO 4217.</ResponseField>
<ResponseField name="data[].billing_period" type="string">`month`, `year`, etc.</ResponseField>
<ResponseField name="data[].current_period_start" type="string|null">ISO 8601.</ResponseField>
<ResponseField name="data[].current_period_end" type="string|null">ISO 8601.</ResponseField>
<ResponseField name="data[].trial_ends_at" type="string|null">ISO 8601.</ResponseField>
<ResponseField name="data[].canceled_at" type="string|null">ISO 8601.</ResponseField>
<ResponseField name="data[].ended_at" type="string|null">ISO 8601.</ResponseField>
<ResponseField name="data[].cancel_at" type="string|null">ISO 8601.</ResponseField>
