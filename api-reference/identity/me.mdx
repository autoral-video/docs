---
title: "Token info"
api: "GET /v1/me"
description: "Retorna informações do token atual, tenant e plano."
---

Endpoint para validar token, descobrir escopos ativos e ver limites do plano. **Não requer escopo** — qualquer token autenticado pode chamar.

## Headers

<ParamField header="Authorization" type="string" required>
  `Bearer aut_...`
</ParamField>

## Response

<ResponseField name="data.token" type="object">
  <Expandable title="propriedades">
    <ResponseField name="id" type="integer">ID interno do token.</ResponseField>
    <ResponseField name="name" type="string">Nome do token.</ResponseField>
    <ResponseField name="description" type="string|null">Descrição livre.</ResponseField>
    <ResponseField name="abilities" type="array<string>">Escopos do token.</ResponseField>
    <ResponseField name="last_used_at" type="string|null">ISO 8601.</ResponseField>
    <ResponseField name="expires_at" type="string|null">ISO 8601.</ResponseField>
    <ResponseField name="created_at" type="string">ISO 8601.</ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="data.tenant" type="object">
  <Expandable title="propriedades">
    <ResponseField name="id" type="integer">ID do tenant.</ResponseField>
    <ResponseField name="name" type="string">Nome.</ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="data.plan" type="object">
  <Expandable title="propriedades">
    <ResponseField name="slug" type="string">`free`, `lite`, `pro`, `scale`.</ResponseField>
    <ResponseField name="name" type="string">Nome.</ResponseField>
    <ResponseField name="api_rate_limit_per_minute" type="integer">Requests/minuto.</ResponseField>
    <ResponseField name="max_api_tokens" type="integer">Limite total de tokens.</ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="data.limits.remaining_api_tokens" type="integer">
  Quantos tokens ainda podem ser criados.
</ResponseField>
