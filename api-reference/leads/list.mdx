---
title: "Listar leads"
api: "GET /v1/leads"
description: "Lista paginada (cursor) de leads capturados."
---

**Escopo:** `leads:read`

## Query params

<ParamField query="per_page" type="integer" default="25">1-100.</ParamField>
<ParamField query="cursor" type="string">Cursor.</ParamField>
<ParamField query="status" type="string">`pending`, `abandoned`, `completed`.</ParamField>
<ParamField query="checkout_id" type="integer">Filtra por checkout.</ParamField>

## Response

<ResponseField name="data[].id" type="integer">ID.</ResponseField>
<ResponseField name="data[].checkout_id" type="integer">FK do checkout.</ResponseField>
<ResponseField name="data[].name" type="string">Nome.</ResponseField>
<ResponseField name="data[].email" type="string">E-mail.</ResponseField>
<ResponseField name="data[].phone" type="string|null">Telefone.</ResponseField>
<ResponseField name="data[].payment_method" type="string|null">Método selecionado.</ResponseField>
<ResponseField name="data[].status" type="string">Status atual.</ResponseField>
<ResponseField name="data[].converted_at" type="string|null">ISO 8601.</ResponseField>
