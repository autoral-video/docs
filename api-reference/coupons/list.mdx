---
title: "Listar cupons"
api: "GET /v1/coupons"
description: "Lista paginada (length-aware) de cupons."
---

**Escopo:** `coupons:read`

## Query params

<ParamField query="per_page" type="integer" default="25">1-100.</ParamField>
<ParamField query="page" type="integer" default="1">Página.</ParamField>
<ParamField query="active" type="boolean">Filtra ativos/inativos.</ParamField>

## Response

<ResponseField name="data[].id" type="integer">ID.</ResponseField>
<ResponseField name="data[].code" type="string">Código (uppercase).</ResponseField>
<ResponseField name="data[].name" type="string|null">Nome.</ResponseField>
<ResponseField name="data[].discount_type" type="string">`percentage` ou `fixed`.</ResponseField>
<ResponseField name="data[].discount_value" type="string">Valor decimal.</ResponseField>
<ResponseField name="data[].usage_limit" type="integer|null">Limite total de uso.</ResponseField>
<ResponseField name="data[].per_customer_limit" type="integer|null">Limite por cliente.</ResponseField>
<ResponseField name="data[].usage_count" type="integer">Quantas vezes foi usado.</ResponseField>
<ResponseField name="data[].single_use" type="boolean">Uso único.</ResponseField>
<ResponseField name="data[].is_active" type="boolean">Ativo.</ResponseField>
<ResponseField name="data[].starts_at" type="string|null">ISO 8601.</ResponseField>
<ResponseField name="data[].expires_at" type="string|null">ISO 8601.</ResponseField>
